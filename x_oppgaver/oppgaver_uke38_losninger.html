function even_sum_loop(n)
    sum = 0
    for i = 1:n
        if i % 2 == 0
            sum += i
        end
    end
    return sum
end

function odd_sum_loop(n)
    sum = 0
    for i = 1:n
        if i % 2 != 0
            sum += i
        end
    end
    return sum
end


function even_sum(n)
    return n * (n + 1)
end

function odd_sum(n)
    return n^2
end

# Testing
n = 10;
