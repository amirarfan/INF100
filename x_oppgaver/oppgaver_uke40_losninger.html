function summerTilN(n)
    sum = 0
    i = 1
    while i <= n
        sum += i
        if sum > 100
            break
        end
        i += 1
    end
    return sum
end

# Eksempel på bruk:
println("Summen er: ", summerTilN(20))

# Oppgave 18
function beregn_npv(initial_investering, årlige_inntekter, diskonteringsrente, år)
    npv = -initial_investering
    for t in 1:år
        npv += årlige_inntekter / (1 + diskonteringsrente)^t
    end
    return npv
end


npv_prosjekt = beregn_npv(1_000_000, 300_000, 0.05, 5)
println("NPV for prosjektet er: $npv_prosjekt")


# Hvis npv_prosjekt er positiv, indikerer det at prosjektet vil gi en positiv avkastning etter å ha tatt hensyn til tidsverdien av penger, og det vil være en god investering.
# Hvis npv_prosjekt er negativ, indikerer det at prosjektet ikke vil dekke kostnadene når man tar hensyn til tidsverdien av penger, og det vil ikke være en god investering.


npv_med_økt_inntekt = beregn_npv(1_000_000, 400_000, 0.05, 5)
println("NPV for prosjektet med økt inntekt er: $npv_med_økt_inntekt")
